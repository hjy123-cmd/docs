Logstash 是日志监控平台的处理核心，从 Kafka 消费原始日志后，通过可配置的 Pipeline 完成格式解析、字段提取、数据脱敏等操作，最终将处理后的结构化数据写入 Elasticsearch。本文介绍 Logstash Pipeline 的配置方式和常用模式。

## Pipeline 基础结构

Logstash Pipeline 由三个阶段组成：

```
Input（输入）→ Filter（过滤/处理）→ Output（输出）
```

每个 Pipeline 对应一个 `.conf` 文件，通常放置在 `/etc/logstash/conf.d/` 目录下。

## 从 Kafka 输入

```ruby /etc/logstash/conf.d/main.conf lines
input {
  kafka {
    # Kafka Broker 列表（替换为实际的 Broker 地址）
    bootstrap_servers => "kafka-host1:9092,kafka-host2:9092,kafka-host3:9092"
    # 订阅所有 logs-* Topic
    topics_pattern => "logs-.*"
    # 消费者组 ID
    group_id => "logstash-consumer-group"
    # 从最新的消息开始消费；首次全量同步时可设置为 earliest
    auto_offset_reset => "latest"
    # 解码 JSON 消息
    codec => json
    # 并行消费线程数
    consumer_threads => 4
    # 消费者轮询超时（ms）
    poll_timeout_ms => 1000
  }
}
```

## 过滤配置

### 基础字段解析

使用 Grok 插件解析日志内容中的时间戳、日志级别、类名等字段：

```ruby /etc/logstash/conf.d/main.conf
filter {
  # 解析 Java 应用日志格式
  # 示例日志：2024-01-15 10:30:00.123 INFO  [OrderService] Order created
  grok {
    match => {
      "message" => "%{TIMESTAMP_ISO8601:log_time} %{LOGLEVEL:level}\\s+\\[%{DATA:logger}\\] %{GREEDYDATA:log_message}"
    }
    # 解析失败时添加标签
    tag_on_failure => ["_grokparsefailure"]
  }

  # 将解析出的时间戳转换为 @timestamp
  if [log_time] {
    date {
      match => ["log_time", "yyyy-MM-dd HH:mm:ss.SSS", "ISO8601"]
      target => "@timestamp"
      remove_field => ["log_time"]
    }
  }

  # 将日志级别统一为大写
  mutate {
    uppercase => ["level"]
  }
}
```

### 敏感数据脱敏

使用 `mutate` 和 `gsub` 对日志中的敏感信息进行脱敏处理：

```ruby /etc/logstash/conf.d/main.conf
filter {
  mutate {
    # 手机号脱敏：138****8888
    gsub => [
      "message", "(1[3-9]\\d)\\d{4}(\\d{4})", "\\1****\\2"
    ]
  }
  mutate {
    # 身份证号脱敏：410***********1234
    gsub => [
      "message", "([0-9]{3})[0-9]{11}([0-9X]{4})", "\\1***********\\2"
    ]
  }
  mutate {
    # 银行卡号脱敏：6222***********1234
    gsub => [
      "message", "([0-9]{4})[0-9]{8,12}([0-9]{4})", "\\1***********\\2"
    ]
  }
}
```

<Warning>
  脱敏规则使用正则表达式，请在测试环境充分验证后再部署到生产环境。错误的脱敏规则可能导致日志数据损坏。
</Warning>

### 字段添加与清理

```ruby /etc/logstash/conf.d/main.conf
filter {
  mutate {
    # 添加处理时间戳
    add_field => { "processed_at" => "%{+YYYY-MM-dd'T'HH:mm:ss.SSSZ}" }
    # 删除不需要存储的内部字段
    remove_field => ["@version", "event", "log", "input"]
  }

  # 根据日志级别添加告警标签
  if [level] == "ERROR" or [level] == "FATAL" {
    mutate {
      add_tag => ["alert"]
    }
  }
}
```

### 条件路由

根据服务名或日志级别将日志路由到不同的 Elasticsearch 索引：

```ruby /etc/logstash/conf.d/main.conf
filter {
  # 为不同服务设置索引名称
  if [service] {
    mutate {
      add_field => { "[@metadata][index_name]" => "logs-%{service}" }
    }
  } else {
    mutate {
      add_field => { "[@metadata][index_name]" => "logs-unknown" }
    }
  }
}
```

## 输出到 Elasticsearch

```ruby /etc/logstash/conf.d/main.conf lines
output {
  elasticsearch {
    # Elasticsearch 集群地址（替换为实际的节点地址）
    hosts => ["http://es-host1:9200", "http://es-host2:9200", "http://es-host3:9200"]
    # 使用元数据中的动态索引名
    index => "%{[@metadata][index_name]}-%{+YYYY.MM.dd}"
    # 认证（如果 ES 启用了安全认证）
    # user => "logstash_writer"
    # password => "${LOGSTASH_ES_PASSWORD}"
    # ILM 策略配置
    ilm_enabled => true
    ilm_rollover_alias => "logs"
    ilm_policy => "logs-default-policy"
    # 批量写入参数
    bulk_max_size => 5000
    # 写入失败时重试
    retry_on_conflict => 3
  }

  # 解析失败的日志写入单独索引，便于排查
  if "_grokparsefailure" in [tags] {
    elasticsearch {
      hosts => ["http://es-host1:9200", "http://es-host2:9200"]
      index => "logs-parse-failed-%{+YYYY.MM.dd}"
    }
  }
}
```

## 完整 Pipeline 示例

```ruby /etc/logstash/conf.d/main.conf lines
input {
  kafka {
    bootstrap_servers => "kafka-host1:9092,kafka-host2:9092,kafka-host3:9092"
    topics_pattern => "logs-.*"
    group_id => "logstash-consumer-group"
    auto_offset_reset => "latest"
    codec => json
    consumer_threads => 4
  }
}

filter {
  grok {
    match => {
      "message" => "%{TIMESTAMP_ISO8601:log_time} %{LOGLEVEL:level}\\s+\\[%{DATA:logger}\\] %{GREEDYDATA:log_message}"
    }
    tag_on_failure => ["_grokparsefailure"]
  }

  if [log_time] {
    date {
      match => ["log_time", "yyyy-MM-dd HH:mm:ss.SSS"]
      target => "@timestamp"
      remove_field => ["log_time"]
    }
  }

  mutate {
    uppercase => ["level"]
    gsub => ["message", "(1[3-9]\\d)\\d{4}(\\d{4})", "\\1****\\2"]
    remove_field => ["@version", "event"]
  }

  # 设置动态索引名；若 [service] 字段不存在则回退到 logs-unknown
  if [service] {
    mutate {
      add_field => { "[@metadata][index_name]" => "logs-%{service}" }
    }
  } else {
    mutate {
      add_field => { "[@metadata][index_name]" => "logs-unknown" }
    }
  }
}

output {
  elasticsearch {
    hosts => ["http://es-host1:9200", "http://es-host2:9200", "http://es-host3:9200"]
    index => "%{[@metadata][index_name]}-%{+YYYY.MM.dd}"
    ilm_enabled => true
    ilm_rollover_alias => "logs"
    ilm_policy => "logs-default-policy"
    bulk_max_size => 5000
  }
}
```
